(defun fatorial(n)
    (if (zerop n)
        1
    (* n (fatorial(1- n)))))

(defun tail_fatorial (number)
  (labels ((sequencia-aux (x acumulador)
             (if (zerop x)
                 acumulador
                 (sequencia-aux (- x 1) (* acumulador x)))))
    (sequencia-aux number 1)))
